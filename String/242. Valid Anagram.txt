class Solution {
    public boolean isAnagram(String s, String t) {
        boolean flag = true;
        if(s.length() != t.length()) return false;
        for(int i = 0;i<s.length();i++) {
            if(t.contains(s.charAt(i)+"")) {
                t = t.replaceFirst(s.charAt(i)+"","");
            }
            else {
                flag = false;
                break;
            }
        }
        return flag;
    }
}
