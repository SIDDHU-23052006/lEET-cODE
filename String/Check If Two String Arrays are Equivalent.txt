class Solution {
    public boolean arrayStringsAreEqual(String[] word1, String[] word2) {
        String s = new String();
        for(String str1:word1) {
            s += str1;
        }
        String s1 = new String();
        for(String str2:word2) {
            s1 +=str2;
        }
        return s.equals(s1);
    }
}
