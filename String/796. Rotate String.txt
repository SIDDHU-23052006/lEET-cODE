class Solution {
    public boolean rotateString(String s, String goal) {
        for(int i = 0;i<s.length();i++){
            char temp = s.charAt(0);
            s = s.replaceFirst(temp+"","");
            s = s+""+temp;
            if(s.equals(goal)) {
                return true;
            }
        }
        return false;
    }
}
