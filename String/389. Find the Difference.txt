class Solution {
    public char findTheDifference(String s, String t) {
        for(int i = 0;i<s.length();i++) {
            // for(int j = 0;j<t.length();j++) {
            //     if(s.charAt(i) == t.charAt(j)) {
            //         t = t.replaceFirst(s.charAt(i)+"","");
            //     }
            // }
            t = t.replaceFirst(s.charAt(i)+"","");
        }
        return t.charAt(0);
    }
}
