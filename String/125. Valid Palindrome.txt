class Solution {
    public boolean isPalindrome(String s) {
        boolean flag = true;
        s = s.toLowerCase();
        int i = 0;
        while(i < s.length()) {
            if(!((s.charAt(i) >= 97 && s.charAt(i) <= 122) || (s.charAt(i) >= 48 && s.charAt(i) <= 57))) {
                s = s.replace(s.charAt(i)+"","");
            }
            else {
                i++;
            }
        }
        i -= 1;
        for(int j = 0;j<s.length();j++) {
            if(s.charAt(i)!=s.charAt(j)) {
                flag = false;
                break;
            }
            i--;
        }
        return flag;
    }
}
