class Solution {
    public int[] getConcatenation(int[] nums) {
        int[] arr = new int[nums.length*2];
        int i = 0;
        int j = 0;
        while(i<nums.length*2){
            if(i < nums.length) {
                arr[i] = nums[i];
            }
            else {
                arr[i] = nums[j];
                j++;
            }
            i++;
        }
        return arr;
    }
}
