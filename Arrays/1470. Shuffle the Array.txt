class Solution {
    public int[] shuffle(int[] nums, int n) {
        int mid = n;
        int[] arr = new int[n*2];
        int i = 0;
        int nor = 0;
        while(i < nums.length) {
            if(i%2==1) {
                arr[i] = nums[mid++];
                i++;
            }
            else {
                arr[i] = nums[nor++];
                i++;
            }
        }
        return arr;
    }
}
