class Solution {
    public int differenceOfSum(int[] nums) {
        int sum1= 0;
        int sum2 = 0;
        for(int i = 0;i<nums.length;i++) {
            sum1+=nums[i];
            int num = nums[i];
            while(num > 0) {
                int temp = num % 10;
                sum2+=temp;
                num/=10;
            }
        }
        return Math.abs(sum1 - sum2);
    }
}
