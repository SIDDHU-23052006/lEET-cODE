class Solution {
    public int addDigits(int num) {
        if(num == 0) {
            return 0;
        }
        while(num/10 != 0) {
            int res = 0;
            int temp = num;
            while(temp > 0) {
                res += temp%10;
                temp/=10;
            }
            num = res;
        }
        return num;
    }
}
