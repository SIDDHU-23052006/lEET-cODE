class Solution {
    public boolean isPalindrome(int x) {
        if(x < 0) {
            return false;
        }
        long check = x;
        long temp = x;
        long count = -1;
        while(temp > 0) {
            count++;
            temp/=10;
        }
        long res = 0;
        while(x > 0) {
            res += (x%10) * Math.pow(10,count--);
            x/=10;
        }
        if(check == res) {
            return true;
        }
        return false;
    }
}
