class Solution {
    public int largestAltitude(int[] gain) {
        int[] arr = new int[gain.length+1];
        int j = 0;
        int sum = 0;
        for(int i = 1;i<arr.length;i++) {
            sum+=gain[j++];
            arr[i] = sum;
        }
        int max = 0;
        for(int i = 0;i<arr.length;i++) {
            if(max < arr[i]) {
                max = arr[i];
            }
        }
        return max;
    }
}
